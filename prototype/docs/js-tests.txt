## Install tools globally ##

# become root
su

npm install -g bower
# ( npm install -g grunt-cli )
npm install -g karma-cli
npm install -g protractor
webdriver-manager update
# ( npm install -g yo )

# install Chrome to be able run Karma:
# cd to the download folder or anywhere outside our project

wget -c https://dl.google.com/linux/direct/google-chrome-stable_current_amd64.deb
dpkg -i google-chrome-*.deb
apt-get install -f

# leave root mode
exit



## Install tools inside the webapp project folder and add them as dependencies ##

npm init

npm install karma --save-dev
npm install karma-jasmine karma-chrome-launcher --save-dev
npm install protractor --save-dev
npm install bower --save-dev

bower init

bower install angular --save
bower install angular-bootstrap --save
bower install angular-mocks --save
bower install angular-route --save
bower install angular-xeditable --save
bower install jquery --save
bower install bootstrap --save



## Configure Karma ##

# inside the folder '/test' just follow the dialog after typing:

karma init

# edit the karma.conf.js for further configuration
# be sure to include all needed libraries under 'files'



## Run unit tests with Karma ##

# alternative 1:
karma start test/karma.conf.js

# alternative 2:
# in package.json > scripts > add "test": "node node_modules/karma/bin/karma start test/karma.conf.js",
npm test
